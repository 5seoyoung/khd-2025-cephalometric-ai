#!/bin/bash
echo "🚀 Starting Cephalometric AI Demo..."

# 환경 변수 설정
export DEMO_MODE=1
export STREAMLIT_SERVER_PORT=8501

# 가상환경 활성화 (옵션)
# source venv/bin/activate

# Streamlit 실행
streamlit run src/demo/streamlit_app.py \
    --server.port=8501 \
    --server.address=localhost \
    --theme.base=light \
    --theme.primaryColor=#2E8B57 \
    --theme.backgroundColor=#FFFFFF \
    --theme.secondaryBackgroundColor=#F0FFF0

echo "✅ Demo started at http://localhost:8501"
